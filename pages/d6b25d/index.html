Java数据结构与算法

# 一、数据结构概述

数据结构分类：

  • 线性结构

    数据元素之间存在一对一的线性关系。常见的线性结构有：数组、队列、链表、栈。

    线性结构有两种不同的存储结构，如下：

    • 顺序存储结构

      存储元素是连续的（地址连续）。

    • 链式存储结构

      存储元素不一定连续。

  • 非线性结构

    常见的非线性结构：二维数组、多维数组、广义表、树结构、图结构

# 二、稀疏数组和队列

# 2.1 稀疏数组

​ 当一个数组中大部分元素为 0 或为相同的值时，使用稀疏数组。如图：

第一行表示：原数组有 6 行，7 列，一共有 8 个坐标位置有值。

第二行表示：第 0 行，第 3 列有一个值为 22 。（之后几行与第二行意义类似）

package datastructure;

/**
 * 稀疏数组
 */
public class RareArray {
    public static void main(String[] args) {
        int[][] origin = new int[6][7];
        origin[0][1] = 1;
        origin[1][2] = 2;
        origin[2][3] = 2;
        System.out.println("原数组👇");
        printArr(origin);
        //找出有元素的位置个数
        int count = 0;
        for (int[] ints : origin) {
            for (int el : ints) {
                if (el != 0) {
                    count++;
                }
            }
        }
        //使用三元组存放稀疏数组
        int[][] rare = new int[count + 1][3];
        int i = 0;
        int j = 0;
        rare[i][j] = 6;
        rare[i][j+1] = 7;
        rare[i][j+2] = count;
        for (int i1 = 0; i1 < origin.length; i1++) {
            for (int i2 = 0; i2 < origin[i1].length; i2++) {
                if (origin[i1][i2] != 0) {
                    rare[++i][j] = i1;
                    rare[i][j+1] = i2;
                    rare[i][j+2] = origin[i1][i2];
                }

            }
        }
        System.out.println("稀疏数组👇");
        printArr(rare);

        //还原
        origin = new int[rare[0][0]][rare[0][1]];
        int len = rare[0][2];
        for (int k = 1; k <= len; k++) {
            origin[rare[k][0]][rare[k][1]] = rare[k][2];
        }
        System.out.println("还原后的数组👇");
        printArr(origin);

    }

    public static void printArr(int[][] arr) {
        for (int[] ints : arr) {
            for (int el : ints) {
                System.out.printf("%d\t",el);
            }
            System.out.println();
        }
    }
}

package datastructure;

import java.util.Arrays;
import java.util.Scanner;

/**
 * 数组实现循环队列
 */
public class ArrayQueue {
    private int maxSize; //数组最大长度
    private int front; //头索引
    private int rear; //尾索引
    private int[] arr; //存数据的arr

    public ArrayQueue(int size) {
        maxSize = size;
        arr = new int[size];
    }

    /**
     * 判空
     *
     * @return
     */
    public boolean isEmpty() {
        return rear == front;
    }

    /**
     * 判满
     *
     * @return
     */
    public boolean isFull() {
        return (rear + 1) % maxSize == front;
    }

    /**
     * 入队
     *
     * @param el
     */
    public void add(int el) {
        if (isFull()) {
            throw new RuntimeException("队列已满!");
        }
//        arr[rear % maxSize] = el;
//        rear++;
        arr[rear] = el;
        rear = (rear + 1) % maxSize;
    }

    /**
     * 出队
     *
     * @return
     */
    public int get() {
        if (isEmpty()) {
            throw new RuntimeException("队列为空！");
        }
        int res = arr[front];
        front = (front + 1) % maxSize;
        return res;
    }

    /**
     * 有效长度
     *
     * @return
     */
    public int size() {
        return (rear - front + maxSize) % maxSize;
    }

    public void print() {
        System.out.println(this.toString());
    }

    public void show() {
        for (int i = front; i < size(); i++) {
            System.out.printf("%d\t", arr[i++]);
        }
    }

    @Override
    public String toString() {
        return "ArrayQueue{" +
                "maxSize=" + maxSize +
                ", front=" + front +
                ", rear=" + rear +
                ", arr=" + Arrays.toString(arr) +
                '}';
    }

    public static void main(String[] args) {
        ArrayQueue queue = new ArrayQueue(4);
        Scanner scanner = new Scanner(System.in);
        while (true) {
            int i = scanner.nextInt();
            if (i == -1) {
                System.out.println(":::" + queue.get());
                continue;
            }
            queue.add(i);

            queue.show();
        }
    }
}

# 3.1 单链表

package datastructure;

/**
 * 单链表
 */
public class LinkedListDemo {
    class Node {
        private int data;
        private Node next;

        Node() {
        }

        Node(int data) {
            this.data = data;
        }

        @Override
        public String toString() {
            return "Node{" +
                    "data=" + data +
                    '}';
        }
    }

    private Node head = new Node();


    /**
     * 添加节点
     * @param node
     */
    public void add(Node node) {
        Node tmp = head;
        while (tmp.next != null) {
            tmp = tmp.next;
        }
        tmp.next = node;
    }

    /**
     * 添加并排序
     *
     * @param node
     */
    public void addByOrder(Node node) {
        Node tmp = head;
        while (tmp.next != null && tmp.next.data < node.data) {
            tmp = tmp.next;
        }
        node.next = tmp.next;
        tmp.next = node;
    }

    /**
     * 打印有效节点
     */
    public void show() {
        Node tmp = head.next;
        if (tmp == null) {
            return;
        }
        while (tmp != null) {
            System.out.println(tmp);
            tmp = tmp.next;
        }
    }

    /**
     * 链表反转
     */
    public void reverse() {
        Node tmp = head.next;
        Node newHead = new Node();
        while (tmp != null) {
            Node m = tmp.next;
            tmp.next = newHead.next;
            newHead.next = tmp;
            tmp = m;
        }
        head = newHead;
    }

    public static void main(String[] args) {
        LinkedListDemo linkedListDemo = new LinkedListDemo();
        linkedListDemo.add(linkedListDemo.new Node(2));
        linkedListDemo.add(linkedListDemo.new Node(1));
        linkedListDemo.add(linkedListDemo.new Node(3));
        linkedListDemo.add(linkedListDemo.new Node(8));
        linkedListDemo.add(linkedListDemo.new Node(5));
        linkedListDemo.add(linkedListDemo.new Node(9));
        linkedListDemo.show();

        System.out.println("===============");

        LinkedListDemo linkedListDemo2 = new LinkedListDemo();
        linkedListDemo2.addByOrder(linkedListDemo.new Node(2));
        linkedListDemo2.addByOrder(linkedListDemo.new Node(1));
        linkedListDemo2.addByOrder(linkedListDemo.new Node(3));
        linkedListDemo2.addByOrder(linkedListDemo.new Node(3));
        linkedListDemo2.addByOrder(linkedListDemo.new Node(8));
        linkedListDemo2.addByOrder(linkedListDemo.new Node(5));
        linkedListDemo2.addByOrder(linkedListDemo.new Node(9));
        linkedListDemo2.show();
        System.out.println("==============");
        linkedListDemo2.reverse();
        linkedListDemo2.show();

    }
}

package datastructure;

import lombok.Data;

/**
 * 约瑟夫问题：
 * n个小朋友绕一圈报数，数到第k个出队，问剩下的是哪一个
 */
public class CircleLinkedList {
    public static void main(String[] args) {
        getCircleList(5,1,2);
    }

    /**
     * n个人，从k开始报数，数到m出列
     *
     * @param n
     * @param k
     * @param m
     * @return
     */
    public static Node getCircleList(int n, int k, int m) {
        Node head = null;
        Node tmp = null;
        //1. 构成单向循环链表
        for (int i = 1; i <= n; i++) {
            if (i == 1) {
                head = new Node();
                head.setData(i);
                tmp = head;
            } else {
                tmp.setNext(new Node());
                tmp = tmp.getNext();
                tmp.setData(i);
            }
        }
        System.out.println(head);
        tmp.setNext(head);

        //2.从第k个开始
        for (int i = 0; i < k - 1; i++) {
            head = head.getNext();
        }
        System.out.println(head.getData());
        while (true) {
            if (head == head.getNext()) {
                break;
            }
            //3.数到m的前一个
            for (int i = 1; i < m - 1; i++) {
                head = head.getNext();
            }

            //4.出列m
            System.out.println("出："+head.getNext().getData());
            head.setNext(head.getNext().getNext());
            //5.下一轮从m的下一个开始数
            head = head.getNext();

        }
        System.out.println(head.getData());
        return head;
    }
}

# 4.5 中缀转后缀并求值

package datastructure;

import java.util.*;

/**
 * 中缀转后缀并求值
 */
public class Expression {
    private static Map<String, Integer> map = new HashMap<>();

    static {
        map.put("+", 1);
        map.put("-", 1);
        map.put("*", 2);
        map.put("/", 2);
    }

    public static void main(String[] args) {
        String express = "3*2+5*3";
        //+*
        //32*53*+
        Stack<String> numStack = new Stack<>();
        Stack<String> operatorStack = new Stack<>();
        List<String> expressList = getExpressList(express);
        for (int i = 0; i < expressList.size(); i++) {
            String s = expressList.get(i);
            if (isOperator(s)) {
                if (!operatorStack.empty() && map.get(s) < map.get(operatorStack.peek())) {
                    numStack.add(operatorStack.pop());
                }
                operatorStack.add(s);
            } else {
                numStack.add(s);
            }
        }

        while (!operatorStack.empty()) {
            numStack.add(operatorStack.pop());
        }

        String[] arr = numStack.toArray(new String[0]);
        System.out.println(Arrays.toString(arr));
        Stack<Integer> res = new Stack<>();
        for (int i = 0; i < arr.length; i++) {
            if (isOperator(arr[i])) {
                cal(arr[i],res);
            } else {
                res.add(Integer.valueOf(arr[i]));
            }
        }
        System.out.println(res.pop());
    }

    private static void cal(String s, Stack<Integer> res) {
        Integer num1 = res.pop();
        Integer num2 = res.pop();
        switch (s) {
            case "+":
                res.add(num1 + num2);
                break;
            case "-":
                res.add(num2 - num1);
                break;
            case "*":
                res.add(num1*num2);
                break;
            case "/":
                res.add(num2/num1);
                break;
            default:
                break;
        }
    }


    private static List<String> getExpressList(String express) {
        List<String> list = new ArrayList<>();
        String[] split = express.split("");
        StringBuffer sb = new StringBuffer();
        for (int i = 0; i < split.length; i++) {
            if (sb.length() > 0) {
                list.add(sb.toString());
                sb = new StringBuffer();
            }
            if (isOperator(split[i])) {
                list.add(split[i]);
            } else {
                sb.append(split[i]);
            }
        }
        if (sb.length() > 0) {
            list.add(sb.toString());
        }
        return list;
    }

    private static boolean isOperator(String s) {
        return "+".equals(s) || "-".equals(s) || "*".equals(s) || "/".equals(s);
    }
}

package datastructure;

/**
 * 迷宫问题
 */
public class QMaze {
    public static void main(String[] args) {
        int[][] map = new int[7][7];
        for (int i = 0; i < map.length; i++) {
            map[i][0] = 1;
            map[i][6] = 1;
            map[0][i] = 1;
            map[6][i] = 1;
        }
        map[3][1] = 1;
        map[3][2] = 1;
        printArr(map);

        System.out.println("==============");
        mazeOut(map, 1, 1);
        printArr(map);
    }

    private static boolean mazeOut(int[][] map, int i, int j) {
        if (map[5][5] == 2) {
            return true;
        } else {
            if (map[i][j] == 0) {
                map[i][j] = 2;
                //下右上左
                if (mazeOut(map, i + 1, j)
                        || mazeOut(map, i, j + 1)
                        || mazeOut(map, i - 1, j)
                        || mazeOut(map, i, j - 1)
                ) {
                    return true;
                } else {
                    map[i][j] = 3;
                    return false;
                }
            } else {
                return false;
            }
        }
    }

    public static void printArr(int[][] map) {
        for (int i = 0; i < map.length; i++) {
            for (int j = 0; j < map[i].length; j++) {
                System.out.printf("%d\t", map[i][j]);
            }
            System.out.println();
        }
    }
}

# 5.2 八皇后问题（92 种）

package datastructure;

import java.util.Arrays;

/**
 * 8皇后问题
 */
public class Queen8 {
    public static int count = 0;

    public static void main(String[] args) {
        int[] arr = new int[8];
        run(arr,0);
        System.out.println(count);
        //1.从第一行第一列开始放，如果与之前的冲突，则回溯，否则继续下一个
    }

    public static void run(int[] arr,int n) {
        if (n == 8) {
            System.out.println(Arrays.toString(arr));
            count++;
            return;
        }
        for (int k = 0; k < arr.length; k++) {
            arr[n] = k;
            if (judge(arr,n)) {
                run(arr,n+1);
            }
        }
    }

    /**
     * 判断是否满足条件：不在同一行，不在同一列
     * @param arr
     * @param n
     * @return
     */
    private static boolean judge(int[] arr, int n) {
        for (int i = 0; i < n; i++) {
            if (arr[i] == arr[n] || Math.abs(n - i) == Math.abs(arr[i] - arr[n])) {
                return false;
            }
        }
        return true;
    }
}

# 6.1 冒泡排序

    /**
     * 冒泡排序：i代表第i趟排序，j代表第j个元素，j+1代表第j+1个元素，
     * 始终是j和j+1位置上的元素在比较大小，每一趟会把最大数移到数组末尾。
     *
     * @param arr
     */
    public static void bubble(int[] arr) {
        System.out.println(Arrays.toString(arr));
        boolean flag = false;
        for (int i = 0; i < arr.length - 1; i++) {
            flag = false;
            for (int j = 0; j < arr.length - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    flag = true;
                    arr[j] ^= arr[j + 1];
                    arr[j + 1] ^= arr[j];
                    arr[j] ^= arr[j + 1];
                }
            }
            if (!flag) {
                break;
            }
            System.out.println(i + "::::" + Arrays.toString(arr));
        }
    }

# 6.2 选择排序

    /**
     * 简单选择，i代表开始挑选第i个位置的元素，记录最小值及其索引
     *
     * @param arr
     */
    public static void selectSort(int[] arr) {
        int index = -1;
        int min;
        System.out.println(Arrays.toString(arr));
        for (int i = 0; i < arr.length - 1; i++) {
            min = arr[i];
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[j] < min) {
                    index = j;
                    min = arr[j];
                }
            }
            if (index != -1) {
                arr[index] = arr[i];
                arr[i] = min;
            }
            System.out.println(index + "::::" + Arrays.toString(arr));
        }
    }

# 6.3 插入排序

    /**
     * 插入排序：i代表要插入的数的索引，while循环用来将满足条件的元素后移
     *
     * @param arr
     */
    public static void insertSort(int[] arr) {
        System.out.println(Arrays.toString(arr));
        for (int i = 1; i < arr.length; i++) {
            int val = arr[i];
            int index = i - 1;
            while (index >= 0 && val < arr[index]) {
                arr[index + 1] = arr[index];
                index--;
            }

            arr[index + 1] = val;
            System.out.println(i + "::::" + Arrays.toString(arr));
        }
    }
