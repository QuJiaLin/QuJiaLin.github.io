    private int front; //å¤´ç´¢å¼•

    private int[] arr; //å­˜æ•°æ®çš„arr

        maxSize = size;
        arr = new int[size];
    }

    /**
     * åˆ¤ç©º

    public boolean isEmpty() {
        return rear == front;
    }

    /**
     * åˆ¤æ»¡

     * @return

    public boolean isFull() {
        return (rear + 1) % maxSize == front;
    }

    /**
     * å…¥é˜Ÿ
     *
     * @param el
     */
    public void add(int el) {

            throw new RuntimeException("é˜Ÿåˆ—å·²æ»¡!");
        }
//        arr[rear % maxSize] = el;

        rear = (rear + 1) % maxSize;
    }

    /**
     * å‡ºé˜Ÿ
     *
     * @return
     */
    public int get() {
        if (isEmpty()) {
            throw new RuntimeException("é˜Ÿåˆ—ä¸ºç©ºï¼");
        }
        int res = arr[front];
        front = (front + 1) % maxSize;
        return res;
    }

    /**
     * æœ‰æ•ˆé•¿åº¦
     *
     * @return
     */
    public int size() {
        return (rear - front + maxSize) % maxSize;
    }

    public void print() {
        System.out.println(this.toString());
    }

    public void show() {
        for (int i = front; i < size(); i++) {
            System.out.printf("%d\t", arr[i++]);
        }
    }

    @Override
    public String toString() {
        return "ArrayQueue{" +
                "maxSize=" + maxSize +
                ", front=" + front +
                ", rear=" + rear +
                ", arr=" + Arrays.toString(arr) +
                '}';
    }

public class Expression {
    private static Map<String, Integer> map = new HashMap<>();

    static {
        map.put("+", 1);
        map.put("-", 1);
        map.put("*", 2);
        map.put("/", 2);
    }

    public static void main(String[] args) {
        String express = "3*2+5*3";
        //+*
        //32*53*+
        Stack<String> numStack = new Stack<>();
        Stack<String> operatorStack = new Stack<>();
        List<String> expressList = getExpressList(express);
        for (int i = 0; i < expressList.size(); i++) {
            String s = expressList.get(i);
            if (isOperator(s)) {
                if (!operatorStack.empty() && map.get(s) < map.get(operatorStack.peek())) {
                    numStack.add(operatorStack.pop());
                }
                operatorStack.add(s);
            } else {
                numStack.add(s);
            }
        }

        while (!operatorStack.empty()) {
            numStack.add(operatorStack.pop());
        }

        String[] arr = numStack.toArray(new String[0]);
        System.out.println(Arrays.toString(arr));
        Stack<Integer> res = new Stack<>();
        for (int i = 0; i < arr.length; i++) {
            if (isOperator(arr[i])) {
                cal(arr[i],res);
            } else {
                res.add(Integer.valueOf(arr[i]));
            }
        }
        System.out.println(res.pop());
    }

    private static void cal(String s, Stack<Integer> res) {
        Integer num1 = res.pop();
        Integer num2 = res.pop();
        switch (s) {
            case "+":
                res.add(num1 + num2);
                break;
            case "-":
                res.add(num2 - num1);
                break;
            case "*":
                res.add(num1*num2);
                break;
            case "/":
                res.add(num2/num1);
                break;
            default:
                break;
        }
    }


    private static List<String> getExpressList(String express) {
        List<String> list = new ArrayList<>();
        String[] split = express.split("");
        StringBuffer sb = new StringBuffer();
        for (int i = 0; i < split.length; i++) {
            if (sb.length() > 0) {
                list.add(sb.toString());
                sb = new StringBuffer();
            }
            if (isOperator(split[i])) {
                list.add(split[i]);
            } else {
                sb.append(split[i]);
            }
        }
        if (sb.length() > 0) {
            list.add(sb.toString());
        }
        return list;
    }

    private static boolean isOperator(String s) {
        return "+".equals(s) || "-".equals(s) || "*".equals(s) || "/".equals(s);
    }
}
